\section{Introduction}

\begin{frame}{Introduction \& Background}
    \begin{block}{\faInfoCircle\ Background}
        In the contemporary professional landscape, the recruitment process serves as a vital bridge between human talent and organizational growth. Traditional job boards and manual recruitment processes are increasingly fragmented and inefficient.
    \end{block}
    
    \vspace{0.3cm}
    \begin{itemize}
        \item \faSearch\ Job seekers navigate dozens of platforms.
        \item \faFileAlt\ Reliance on static, manual resume formatting.
        \item \faUsers\ Recruiters are overwhelmed by unstandardized applications.
    \end{itemize}
\end{frame}

\section{Problem Statement}

\begin{frame}{Problem Statement}
    The existing recruitment methodology presents key pain points:
    
    \vspace{0.5cm}
    \begin{columns}[T,onlytextwidth]
        \column{.48\textwidth}
            \begin{block}{\faUserTie\ For Job Seekers}
                \begin{itemize}
                    \item Inadequacy of static resumes.
                    \item Fragmented public sector (Govt.) listings.
                    \item Lack of interactive career guidance.
                \end{itemize}
            \end{block}

        \column{.48\textwidth}
            \begin{block}{\faBuilding\ For Employers}
                \begin{itemize}
                    \item High screening overhead.
                    \item Cognitive fatigue and bias during evaluation.
                    \item Extended hiring cycles.
                \end{itemize}
            \end{block}
    \end{columns}
\end{frame}

\section{Literature Review}

\begin{frame}{Literature Review}
    \begin{itemize}
        \item \textbf{AI Recommendation Systems:} Significantly increase job matching accuracy by analyzing user profiles, education, and skills.
        \vspace{0.2cm}
        \item \textbf{NLP in Recruitment:} Conversational agents and LLMs emerge as powerful tools for resume parsing and conducting initial candidate interviews.
        \vspace{0.2cm}
        \item \textbf{Serverless Architectures:} Cloud-based platforms handle unpredictable traffic spikes cost-effectively with zero server management overhead.
    \end{itemize}
    \vspace{0.3cm}
    \begin{alertblock}{Our Innovation}
        JobNova bridges these domains by combining local LLM capabilities with edge serverless execution to create a fast, intelligent platform.
    \end{alertblock}
\end{frame}

\section{Proposed Solution}

\begin{frame}{JobNova: The Proposed Solution}
    An AI-driven, serverless recruitment platform tailored to modernize hiring using local Large Language Models (LLMs) and edge databases.
    
    \vspace{0.3cm}
    \begin{description}
        \item[\faRobot\ \textbf{Conversational AI Onboarding}] Replaces dense forms with a chat-based profile builder.
        \item[\faComments\ \textbf{AI Job Agent}] Personalized career assistant for recommendations.
        \item[\faClipboardCheck\ \textbf{Automated Screening}] Evaluates and scores candidates through short interactive chats.
        \item[\faLandmark\ \textbf{Unified Jobs Portal}] Aggregates private, government, and MSME opportunities.
    \end{description}
\end{frame}

\section{System Objectives}

\begin{frame}{Key Objectives}
    \begin{enumerate}
        \item \textbf{Automate Resume Screening:} Grade candidate answers against specific employer prompts.
        \item \textbf{Simplify Profile Construction:} Extract and structure history into standard databases via chat.
        \item \textbf{Aggregate Government \& Private Jobs:} Consolidate jobs into a single unified search portal.
        \item \textbf{Interactive Career Support:} Provide an AI agent for advice and mock interviews.
        \item \textbf{Serverless Architecture:} Build a high-performance system with zero-cold-start functions.
    \end{enumerate}
\end{frame}

\section{System Architecture}

\begin{frame}{System Architecture Overview}
    JobNova utilizes a highly scalable, low-latency layered serverless architecture:
    \vspace{0.2cm}
    \begin{block}{\faServer\ Architecture Layers}
    \begin{itemize}
        \item \textbf{Frontend Layer:} Nuxt 4, Vue 3, Tailwind CSS.
        \item \textbf{Edge Logic Layer:} Cloudflare Workers \& Pages (Nitro).
        \item \textbf{Storage Layer:} Cloudflare D1 (SQL) and R2 Bucket.
        \item \textbf{AI Layer:} Local Ollama Framework (Granite-350m model).
    \end{itemize}
    \end{block}
    
    \vspace{0.3cm}
    \cookiekicker{\faLightbulb\ Decoupled Processing}
    \vspace{0.2cm}
    
    \textit{Decoupling core web services from AI inference ensures LLM tasks do not block standard web requests.}
\end{frame}
